    def initialize(data_dir)
      fmis = FileMappedInputStream.new(data_dir + "/matrix.bin")
      @left_size = fmis.get_int
      @right_size = fmis.get_int
      @matrix = fmis.get_short_array(@left_size * @right_size)
      fmis.close
    end
