    def self.read_categories(data_dir)
      data = FileMappedInputStream::get_int_array(data_dir + "/char.category")
      size = data.size / 4
      ary = []
      for i in 0 .. (size - 1)
        ary.push(Category.new(data[i * 4], data[i * 4 + 1], data[i * 4 + 2] == 1, data[i * 4 + 3] == 1))
      end
      return ary
    end
